#Usage

	chiconv < infile > outfile
	chiconv -i infile infile2 infile3 #Save in-place if no error
	chiconv -f infile infile2 infile3 #Save in-place regardless of errors

#Support encoding
	UTF-8
	Big5 (UAO)
	Big5 (HKSCS 2004)
	Big5 (HKSCS 2001)
	Big5 (HKSCS 1999)
	Big5-2003
	Big5e
	Big5-ETEN
	GBK
	CCCII
	UTF-16LE
	UTF-16BE
	UTF-32LE
	UTF-32BE
	GB18030
